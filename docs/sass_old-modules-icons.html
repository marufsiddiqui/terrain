-- Icons -- //

Use the following CSS code if you want to use data attributes for inserting your icons:

@font-face {
  font-family: 'icomoon';
  src: url('../fonts/icons/icomoon.eot');
  src: url('../fonts/icons/icomoon.eot?#iefix') format('embedded-opentype'),
       url('../fonts/icons/icomoon.woff') format('woff'),
       url('../fonts/icons/icomoon.ttf') format('truetype'),
       url('../fonts/icons/icomoon.svg#icomoon') format('svg');
  font-weight: normal;
  font-style: normal;
}

[data-icon]:before { content: attr(data-icon); }

[data-icon]:before, [class*="icon-"]:before, %icon:before {
  font-family: 'icomoon';
  speak: none;
  font-style: normal;
  font-weight: normal;
  font-variant: normal;
  text-transform: none;
  line-height: 1;
  -webkit-font-smoothing: antialiased;
  margin-right: 0.5em;
}

.icon-in-circle {
  position: relative;
  padding-left: $line-height * 1.5;
  &:before {
    position: absolute;
    top: 0;
    left: 0;
    text-align: center;
    width: $line-height; height: $line-height;
    line-height: $line-height;
    @include border-radius(50%);
  }
}

.icon-only {
  display: inline-block;
  vertical-align: middle;
  position: relative;
  text-indent: 9999em;
  width: $line-height; height: $line-height;
  line-height: $line-height;
  white-space: nowrap;
  overflow: hidden;

  &:before {
    position: absolute;
    top: 0;
    left: 0; right: 0;
    text-align: center;
    height: $line-height;
    line-height: $line-height;
    text-indent: 0;
  }
}
