  -webkit-appearance: none;
  width: 100%;
  padding: $line-height / 3;
  line-height: $line-height - $line-height / 4;
  background: transparent;
  outline: 0;
  box-sizing: border-box;
  box-shadow: none;
  font-family: $sans-stack;
  font-size: $font-size;
  font-weight: 300;
  border-size: 0.2em;
  border-width: 0.2em;
  margin-top: 0;
  margin-bottom: $line-height;
  border-style: solid;
  @include box-sizing(border-box);
  @include border-radius($radius);
  @include transition( border-color 0.5s);

    @extend %black_transparent_primary;
  }
  &:focus {
    @extend %black_transparent_light-secondary;
  }
}
